import turtle

# Setup the screen and turtle object
screen = turtle.Screen()
screen.bgcolor("white")
flower = turtle.Turtle()
flower.speed(10)  # Set drawing speed (1 is slow, 10 is fast, 0 is fastest)
flower.color("purple")
flower.shape("turtle")

def draw_petal():
    """Draws a single curved petal using two half-circles."""
    flower.circle(100, 60) # Draw an arc of 60 degrees with a radius of 100
    flower.left(120)       # Turn left to reposition for the second arc
    flower.circle(100, 60) # Draw the second arc
    flower.left(120)       # Turn left to face the original direction

# Draw the petals
for _ in range(6):
    draw_petal()
    flower.left(60) # Rotate to draw the next petal

# Draw the flower center
flower.penup()
flower.goto(0, -40) # Move to the center bottom of the flower
flower.pendown()
flower.color("yellow")
flower.begin_fill()
flower.circle(40) # Draw a filled circle for the center
flower.end_fill()

# Finalize the drawing
flower.hideturtle()
screen.exitonclick()
